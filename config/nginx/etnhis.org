# Redirect HTTP to HTTPS for admin site
server {
	listen 80;
	server_name etnhis.org www.etnhis.org;
	return 301 https://etnhis.org$request_uri;
}

# Redirect HTTP to HTTPS for API
server {
	listen 80;
	server_name api.etnhis.org;
	return 301 https://$host$request_uri;
}

# Redirect HTTP to HTTPS for Metabase
server {
	listen 80;
	server_name metabase.etnhis.org;
	return 301 https://metabase.etnhis.org$request_uri;
}

server {
	listen [::]:443 ssl ipv6only=on; # managed by Certbot
	listen 443 ssl; # managed by Certbot
	ssl_certificate /etc/letsencrypt/live/etnhis.org/fullchain.pem; # managed by Certbot
	ssl_certificate_key /etc/letsencrypt/live/etnhis.org/privkey.pem; # managed by Certbot
	include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
	ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

	root /home/pilotadmin/build;

	# Add index.php to the list if you are using PHP
	index index.html index.htm index.nginx-debian.html;

	server_name etnhis.org;

	location / {
		# First attempt to serve request as file, then
		# as directory, then fall back to displaying a 404.
		try_files $uri $uri/ /index.html;
	}
}

  }
